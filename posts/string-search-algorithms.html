Brute Force

n length of string
m length of pattern

search at all position starting from 0 to n-m.

void Bf(char *x, int m, char *y, int n) {
    int i, j;

    for(j=0;j<=n-m;++j) {
        for(i=0;i<m && x[i] == y[j+i];++i) {
            if(i>=m)
                return j;
        }
    }
}

// A more efficient implementation

void Bf(char *x, int m, char *y, int n) {
    char *yb;

    for(yb = y; *y != '/0', yb++) {
        if(memcmp(x, yb, m) == 0)
            return y-yb;
    }
}
