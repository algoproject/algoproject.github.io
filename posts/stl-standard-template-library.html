Advanced usage

class MeanValue {
    long num;
    long sum;
    public:
        MeanValue() : num(0), sum(0) {}

        void operator() (int ele) {
            ++num;
            sum += ele;
        }

        double value() {
            return static_cast<double> sum/num;
        }
};

int main() {
    vector<int> col = {1,2,3};
    MeanValue mv = for_each(col.begin(), col.end(), MeanValue());
    cout << mv.value() << endl;
}
